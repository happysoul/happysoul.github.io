The Code

	This following code will assume you've already strobed/reset $4016:

; after code is done, key will be in A
;--CODE START--;
ReadKeyByte:
	ldx #8
loopKeybyte:
	ora $4016
	asl
	dex
	bne loopKeybyte
;--CODE END--;
